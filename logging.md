### SwiftUI

```swift

class Author {
  var firstName: String
  var lastName: String?
  
  init(first: String, last: String?) {
    firstName = first
    lastName = last
  }
}

var author = Author(first: "Paul", last: nil)

// Log to console
print(author)


```
